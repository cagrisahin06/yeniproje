// DCT Watermark Embed Kernel
__global__ void dctWatermarkKernel(
    float* dctCoeffs,
    const WatermarkPattern* pattern,
    int width, int height,
    float strength
) {
    int x = blockIdx.x * blockDim.x + threadIdx.x;
    int y = blockIdx.y * blockDim.y + threadIdx.y;
    
    if (x >= width || y >= height) return;
    
    int idx = y * width + x;
    float coeff = dctCoeffs[idx];
    
    // HVS-aware embedding
    float freqWeight = sqrtf((float)(x*x + y*y)) / 
                       sqrtf(width*width + height*height);
    float adaptiveStrength = strength * (0.5f + 0.5f * freqWeight);
    
    // QIM (Quantization Index Modulation)
    float quantStep = QSTEP_BASE * (1.0f + freqWeight);
    int quantIndex = (int)(coeff / quantStep);
    int bit = pattern->bits[idx % pattern->length];
    
    dctCoeffs[idx] = (quantIndex + bit + 0.5f) * quantStep;
}
// Haar DWT Kernel
__global__ void haarDWTKernel(
    float* input,
    float* ll, float* lh, float* hl, float* hh,
    int width, int height
) {
    int x = blockIdx.x * blockDim.x + threadIdx.x;
    int y = blockIdx.y * blockDim.y + threadIdx.y;
    
    if (x >= width/2 || y >= height/2) return;
    
    int i00 = (2*y) * width + (2*x);
    int i01 = (2*y) * width + (2*x + 1);
    int i10 = (2*y + 1) * width + (2*x);
    int i11 = (2*y + 1) * width + (2*x + 1);
    
    float a = input[i00], b = input[i01];
    float c = input[i10], d = input[i11];
    
    ll[y*(width/2)+x] = (a + b + c + d) / 4.0f;
    lh[y*(width/2)+x] = (a + b - c - d) / 4.0f;
    hl[y*(width/2)+x] = (a - b + c - d) / 4.0f;
    hh[y*(width/2)+x] = (a - b - c + d) / 4.0f;
}

#define GF_SIZE 256
#define RS_N 255
#define RS_K 223

// Galois Field GF(2^8)
static uint8_t rs_gen_poly[RS_N - RS_K + 1];

void rs_encode(uint8_t* data, uint8_t* parity) {
    memset(parity, 0, RS_N - RS_K);
    
    for (int i = 0; i < RS_K; i++) {
        uint8_t feedback = gf_add(data[i], parity[0]);
        if (feedback != 0) {
            for (int j = 0; j < RS_N - RS_K - 1; j++) {
                parity[j] = gf_add(parity[j+1],
                    gf_mul(feedback, rs_gen_poly[j+1]));
            }
            parity[RS_N-RS_K-1] = gf_mul(feedback,
                rs_gen_poly[RS_N-RS_K]);
        } else {
            memmove(parity, parity+1, RS_N-RS_K-1);
            parity[RS_N-RS_K-1] = 0;
        }
    }
